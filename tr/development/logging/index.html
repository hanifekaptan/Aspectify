    Loglama Sistemi¶

    Aspectify, kapsamlı bir loglama sistemi kullanarak uygulama davranışını izler ve hata ayıklama süreçlerini kolaylaştırır.

    Genel Bakış¶

    Loglama sistemi şu özellikleri sağlar:

    • Dosya Tabanlı Loglama: Loglar logs/app.log dosyasına yazılır
    • Yapılandırılabilir Seviyeler: Ortam değişkeni ile kontrol edilebilir
    • UTF-8 Desteği: Türkçe karakter desteği
    • Otomatik Dizin Oluşturma: logs dizini otomatik oluşturulur
    • Timestamp: Her log girişi zaman damgası içerir

    Yapılandırma¶

    Loglama Yapılandırması¶

    app/core/logging_config.py dosyasında tanımlanır:

    import logging
import os
from app.core.config import settings

def setup_logging():
    log_level = settings.LOG_LEVEL.upper()

    # Logs dizini oluştur
    os.makedirs("logs", exist_ok=True)

    # Loglama yapılandırması
    logging.basicConfig(
        level=log_level,
        format='%(asctime)s - %(name)s - %(levelname)s - %(message)s',
        handlers=[
            logging.FileHandler('logs/app.log', encoding='utf-8')
        ]
    )

    logger = logging.getLogger(__name__)
    logger.info(f"Logging initialized with level: {log_level}")
    return logger

logger = setup_logging()

    Ortam Değişkenleri¶

    .env dosyasında log seviyesini ayarlayın:

    # Log seviyesi (DEBUG, INFO, WARNING, ERROR, CRITICAL)
LOG_LEVEL=INFO

    Log Seviyeleri¶

    Seviye Hiyerarşisi¶

    CRITICAL (50) > ERROR (40) > WARNING (30) > INFO (20) > DEBUG (10)

    Seviye Açıklamaları¶

    Seviye Açıklama Kullanım
    DEBUG Hata ayıklama için ayrıntılı bilgi Geliştirme sırasında
    INFO Program yürütmesi hakkında genel bilgi Normal işlemler
    WARNING Beklenmeyen bir şey oldu Potansiyel sorunlar
    ERROR Ciddi bir sorun oluştu Hata durumları
    CRITICAL Çok ciddi bir hata oluştu Sistem çökmesi

    Seviye Kullanımı¶

    from app.core.logging_config import logger

# Hata ayıklama bilgisi
logger.debug("Değişken değeri: %s", variable_value)

# Genel bilgi
logger.info("İşlem başlatıldı: %s", operation_name)

# Uyarı
logger.warning("Beklenmeyen durum: %s", warning_message)

# Hata
logger.error("İşlem başarısız: %s", error_message)

# Kritik hata
logger.critical("Sistem çökmesi: %s", critical_error)

    import time
import logging
from app.core.logging_config import logger

def timed_operation():
    start_time = time.time()

    try:
        # İşlem kodları
        result = perform_operation()

        elapsed_time = time.time() - start_time
        logger.info(f"İşlem tamamlandı: {elapsed_time:.2f} saniye")

        return result

    except Exception as e:
        elapsed_time = time.time() - start_time
        logger.error(f"İşlem başarısız: {elapsed_time:.2f} saniye, hata: {e}")
        raise

    Log Rotation¶

    import logging
from logging.handlers import RotatingFileHandler
from app.core.config import settings

def setup_logging_with_rotation():
    log_level = settings.LOG_LEVEL.upper()

    # Logs dizini oluştur
    os.makedirs("logs", exist_ok=True)

    # Rotating file handler
    handler = RotatingFileHandler(
        'logs/app.log',
        maxBytes=10*1024*1024,  # 10MB
        backupCount=5
    )

    # Formatter
    formatter = logging.Formatter(
        '%(asctime)s - %(name)s - %(levelname)s - %(message)s'
    )
    handler.setFormatter(formatter)

    # Logger yapılandırması
    logger = logging.getLogger()
    logger.setLevel(log_level)
    logger.addHandler(handler)

    return logger

    Log Temizleme¶

    import os
import glob
from datetime import datetime, timedelta

def cleanup_old_logs(days=30):
    """Eski log dosyalarını temizle"""
    cutoff_date = datetime.now() - timedelta(days=days)

    log_files = glob.glob("logs/*.log*")

    for log_file in log_files:
        file_time = datetime.fromtimestamp(os.path.getmtime(log_file))

        if file_time < cutoff_date:
            os.remove(log_file)
            logger.info(f"Eski log dosyası silindi: {log_file}")

    Güvenlik¶

    Hassas Bilgi Maskeleme¶

    import logging
from app.core.logging_config import logger

def mask_sensitive_data(data: str) -> str:
    """Hassas bilgileri maskele"""
    # API anahtarlarını maskele
    if "api_key" in data.lower():
        return data.replace(data, "***MASKED***")

    # Email adreslerini maskele
    import re
    email_pattern = r'\b[A-Za-z0-9._%+-]+@[A-Za-z0-9.-]+\.[A-Z|a-z]{2,}\b'
    return re.sub(email_pattern, "***@***.***", data)

# Kullanım
sensitive_data = "API key: sk-1234567890abcdef"
logger.info(f"API çağrısı: {mask_sensitive_data(sensitive_data)}")

    import pytest
import logging
from app.core.logging_config import logger

def test_logging_setup():
    """Loglama yapılandırmasını test et"""
    assert logger.level == logging.INFO
    assert len(logger.handlers) > 0

def test_log_file_creation():
    """Log dosyasının oluşturulduğunu test et"""
    import os
    assert os.path.exists("logs/app.log")

def test_log_content():
    """Log içeriğini test et"""
    logger.info("Test mesajı")

        content = f.read()
